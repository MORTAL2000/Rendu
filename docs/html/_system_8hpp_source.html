99  void forParallel(size_t low, size_t high, ThreadFunc func){
100  // Make sure the loop is increasing.
101  if(high < low){
102  const size_t temp = low;
103  low = high;
104  high = temp;
105  }
106  // Prepare the threads pool.
107  const size_t count = size_t(std::max(std::thread::hardware_concurrency(), unsigned(1)));
108  std::vector<std::thread> threads;
109  threads.reserve(count);
110 
111  // Compute the span of each thread.
112  size_t span = size_t(std::round((float(high) - float(low))/float(count)));
113  span = std::max(size_t(1), span);
114 
115  // Helper to execute the function passed on a subset of the total interval.
116  auto launchThread = [&func](size_t a, size_t b) {
117  for(size_t i = a; i < b; ++i) {
118  func(i);
119  }
120  };
121 
122  for(size_t tid = 0; tid < count; ++tid){
123  // For each thread, call the same lambda with different bounds as arguments.
124  const size_t threadLow = tid * span;
125  size_t threadHigh = (tid == count - 1) ? high : ((tid+1) * span);
126  threads.emplace_back(launchThread, threadLow, threadHigh);
127  }
128  // Wait for all threads to finish.
129  std::for_each(threads.begin(),threads.end(),[](std::thread& x){x.join();});
130  }
